program for prime number

#include 
 main()
{
   int n, c;
 
   printf("Enter a number\n");
   scanf("%d", &n);
 
   if ( n == 2 )
      printf("Prime number.\n");
   else
   {
       for ( c = 2 ; c <= n - 1 ; c++ )
       {
           if ( n % c == 0 )
              break;
       }
       if ( c != n )
          printf("Not prime.\n");
       else
          printf("Prime number.\n");
   }
   return 0;
}

Fibonacci series

#include 
 
main()
{
   int n, first = 0, second = 1, next, c;
 
   printf("Enter the number of terms\n");
   scanf("%d",&n);
 
   printf("First %d terms of Fibonacci series are :-\n",n);
 
   for ( c = 0 ; c < n ; c++ )
   {
      if ( c <= 1 )
         next = c;
      else
      {
         next = first + second;
         first = second;
         second = next;
      }
      printf("%d\n",next);
   }
 
   return 0;
}

For GCC users

If you are using GCC on Linux operating system then you need to modify programs. For example consider the following program which prints first ten natural numbers 

#include 
#include 
 
int main()
{
    int c;
 
    for ( c = 1 ; c <= 10 ; c++ )
        printf("%d\n", c);
 
    getch();
    return 0;
}
Above source code includes a header file and uses function getch, but this file is Borland specific so it works in turbo c compiler but not in GCC. So the code for GCC should be like #include 
 
int main()
{
    int c;
 
    /* for loop */
 
    for ( c = 1 ; c <= 10 ; c++ )
        printf("%d\n", c);
 
    return 0;
}
