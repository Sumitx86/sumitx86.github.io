void minhand(int t)

{

setcolor(BLACK);

line( X, Y, x1, y1);


setcolor(WBC+1);

line( X, Y, X+ 80 * cos((t-1)*0.1047),Y+ 80 * sin((t-1)*0.1047));

circle(X,Y,4);

}
